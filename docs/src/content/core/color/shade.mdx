# shade

`shade` is a function that generates a color by darkening or lightening a base color.

## Signature



```ts
function shade(color: string, weight: number): string;
```

## Example

```tsx
import { shade } from "@fewings/core/color";

const originColor = "#de4343";
const colors = [
  {
    color: shade(originColor, 0),
    value: 0,
  },
  {
    color: shade(originColor, 30),
    value: 30,
  },
  {
    color: shade(originColor, -30),
    value: -30,
  },
];

const ShadeExample = () => {
  return (
    <div
      style={{
        display: "flex",
        gap: 10,
        justifyContent: "center",
        padding: 40,
      }}
    >
      {colors.map(({ color, value }) => (
        <div
          style={{
            backgroundColor: color,
            color: "white",
            width: 200,
            height: 80,
            fontWeight: "bold",
            display: "flex",
            justifyContent: "center",
            flexDirection: "column",
            alignItems: "center",
          }}
        >
          <div>{color}</div>
          <div
            style={{
              fontSize: 14,
            }}
          >
            shade({originColor},{value})
          </div>
        </div>
      ))}
    </div>
  );
};

export default ShadeExample;
```
